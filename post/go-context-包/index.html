go context 包

2018-09-23
golang

一开始接触 golang 协程的时候，如果想停止子协程，可以通过 channel 通信的方式，主协程通知子协程终止。

package main

import (
	"fmt"
	"time"
)

func test(ch <-chan bool) {
	i, ticker := 0, time.NewTicker(time.Second)
	for _ = range ticker.C {
		select {
		case <-ch:
			return
		default:
			fmt.Println("test", i)
			i++
		}
	}
}

func test1(ch <-chan bool) {
	i, ticker := 0, time.NewTicker(time.Second)
	for _ = range ticker.C {
		select {
		case <-ch:
			return
		default:
			fmt.Println("test1", i)
			i++
		}
	}
}

func main() {
	ch := make(chan bool)
	go test(ch)
	go test1(ch)
	time.Sleep(time.Second * 5)
	ch <- true
	close(ch)
}

以上述代码为例，在主协程睡眠的 5s 内，两个子协程每秒会有一个输出，直到主协程向通道 ch 发送信息并关闭通道后，两个子协程才终止。

但是这会有一个问题，如果我们想通过停止更多的子协程，则在关闭该通道前必须想该通道通信欲关闭的子协程数 - 1次，显然这是件很繁琐的事，这就是 context 包解决的一个痛点。

修改代码：

package main

import (
	"context"
	"fmt"
	"time"
)

func test(ctx context.Context) {
	i, ticker := 0, time.NewTicker(time.Second)
	for _ = range ticker.C {
		select {
		case <-ctx.Done():
			return
		default:
			fmt.Println("test", i)
			i++
		}
	}
}

func test1(ctx context.Context) {
	i, ticker := 0, time.NewTicker(time.Second)
	for _ = range ticker.C {
		select {
		case <-ctx.Done():
			return
		default:
			fmt.Println("test1", i)
			i++
		}
	}
}

func main() {
	ctx, cancelCtx := context.WithCancel(context.Background())
	go test(ctx)
	go test1(ctx)
	time.Sleep(time.Second * 5)
	cancelCtx()
}

当调用cancelCtx方法后，ctx.Done()会返回一个关闭的channel，子协程接收后就会终止，实现更加优雅。
